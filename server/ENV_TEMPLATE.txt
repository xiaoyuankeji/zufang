# 正式上线建议设置 production（后端会禁止使用 sk_test_...，避免误上测试模式）
NODE_ENV=production

# 前端域名（Stripe Checkout success/cancel 会回跳到该地址）
WEB_BASE_URL=https://your-domain.example

# Stripe 后端密钥（必须是 sk_ 开头）
# 正式：sk_live_...
STRIPE_SECRET_KEY=sk_live_xxx

# Stripe Webhook 签名密钥（正式上线必须配置，并且必须是 Live endpoint 对应的 whsec）
STRIPE_WEBHOOK_SECRET=whsec_xxx
